-- ESP Logic
local function createESPBox(player)
    local char = player.Character
    if not char then return end
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hum or hum.Health <= 0 then return end

    -- Box ESP
    local box = Drawing.new("Square")
    box.Color = Color3.fromRGB(255, 0, 0) -- Default red
    box.Thickness = 2
    box.ZIndex = 5
    box.Visible = false
    espBoxes[player] = box

    -- Name ESP
    local nameLabel = Drawing.new("Text")
    nameLabel.Font = "SourceSansBold"
    nameLabel.Size = 18
    nameLabel.Color = Color3.fromRGB(255, 255, 255)
    nameLabel.ZIndex = 5
    nameLabel.Visible = false
    espNames[player] = nameLabel

    -- Skeleton ESP
    local skeletonParts = {}
    for _, part in ipairs(char:GetDescendants()) do
        if part:IsA("Part") and part.Name ~= "HumanoidRootPart" then
            local line = Drawing.new("Line")
            line.Color = Color3.fromRGB(0, 255, 0)
            line.Thickness = 2
            line.ZIndex = 4
            line.Visible = false
            table.insert(skeletonParts, line)
        end
    end
    espSkeletons[player] = skeletonParts
end

local function updateESP()
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= Player and player.Character then
            local char = player.Character
            local hum = char:FindFirstChildOfClass("Humanoid")
            if hum and hum.Health > 0 then
                if not espBoxes[player] then
                    createESPBox(player)
                end

                local box = espBoxes[player]
                local nameLabel = espNames[player]
                local skeleton = espSkeletons[player]

                local hrp = char:FindFirstChild("HumanoidRootPart")
                if hrp then
                    local torso = char:FindFirstChild("Torso") or char:FindFirstChild("UpperTorso")
                    local head = char:FindFirstChild("Head")

                    if torso and head then
                        local torsoPos, onScreenTorso = Camera:WorldToScreenPoint(torso.Position)
                        local headPos, onScreenHead = Camera:WorldToScreenPoint(head.Position)

                        if onScreenTorso and onScreenHead then
                            box.Visible = boxEspEnabled
                            nameLabel.Visible = nameEspEnabled
                            if skeleton then for _, line in ipairs(skeleton) do line.Visible = skeletonEspEnabled end end

                            local width = math.abs(headPos.X - torsoPos.X) * 1.5
                            local height = math.abs(headPos.Y - torsoPos.Y) * 2.2

                            box.Position = Vector2.new(headPos.X - width / 2, headPos.Y - height)
                            box.Size = Vector2.new(width, height)
                            box.Color = if highlightEnemies and player:GetRankInGroup(game.CreatorId) > 0 then Color3.fromRGB(255, 165, 0) else Color3.fromRGB(255, 0, 0) -- Orange for group members, Red otherwise

                            nameLabel.Position = Vector2.new(headPos.X, headPos.Y - height - 15)
                            nameLabel.Text = player.Name .. " [" .. math.floor(hum.Health) .. " HP]"

                            -- Skeleton ESP Update
                            if skeleton and skeletonEspEnabled then
                                local currentSkeletonIndex = 1
                                for _, part in ipairs(char:GetDescendants()) do
                                    if part:IsA("Part") and part.Name ~= "HumanoidRootPart" then
                                        local line = skeleton[currentSkeletonIndex]
                                        if line then
                                            local pos1, onScreen1 = Camera:WorldToScreenPoint(part.Position)
                                            local pos2, onScreen2 = Camera:WorldToScreenPoint(part.Parent.Position)
                                            if onScreen1 and onScreen2 then
                                                line.Visible = true
                                                line.From = Vector2.new(pos1.X, pos1.Y)
                                                line.To = Vector2.new(pos2.X, pos2.Y)
                                            else
                                                line.Visible = false
                                            end
                                        end
                                        currentSkeletonIndex = currentSkeletonIndex + 1
                                    end
                                end
                            end
                        else
                            box.Visible = false
                            nameLabel.Visible = false
                            if skeleton then for _, line in ipairs(skeleton) do line.Visible = false end end
                        end
                    end
                end
            else
                -- Player died or character removed, hide ESP
                if espBoxes[player] then espBoxes[player]:Remove() end
                if espNames[player] then espNames[player]:Remove() end
                if espSkeletons[player] then for _, line in ipairs(espSkeletons[player]) do line:Remove() end end
                espBoxes[player] = nil
                espNames[player] = nil
                espSkeletons[player] = nil
            end
        end
    end
end

-- Aimbot Logic
local function getClosestTarget()
    local closestPlayer = nil
    local closestDist = math.huge
    local screenCenter = Camera.ViewportSize / 2

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= Player and player.Character then
            local char = player.Character
            local hum = char:FindFirstChildOfClass("Humanoid")
            local hrp = char:FindFirstChild("HumanoidRootPart")

            if hum and hum.Health > 0 and hrp then
                local targetPos, onScreen = Camera:WorldToScreenPoint(hrp.Position)
                if onScreen then
                    -- Simple FOV check: distance from screen center
                    local dist = (Vector2.new(targetPos.X, targetPos.Y) - screenCenter):Magnitude()
                    local fovRadius = (fov / 2) * (Camera.ViewportSize.Y / 180) -- Convert FOV degrees to pixels

                    if dist < closestDist and dist < fovRadius then
                        closestDist = dist
                        closestPlayer = player
                    end
                end
            end
        end
    end
    return closestPlayer
end

local function aimAt(target)
    if not target or not target.Character or not target.Character:FindFirstChild("HumanoidRootPart") then return end
    local targetHRP = target.Character.HumanoidRootPart
    local targetPosition = targetHRP.Position

    local aimPart = target.Character:FindFirstChild(targetPart) -- e.g., Head
    if aimPart then
        targetPosition = aimPart.Position
    end

    -- Calculate the target CFrame
    local lookAtCFrame = CFrame.new(HumanoidRootPart.Position, targetPosition)
    local cameraLookVector = Camera.CFrame.LookVector

    -- Smoothly interpolate the camera's CFrame
    local smoothFactor = 1 - aimSpeed -- Lower aimSpeed means faster aiming
    local newCameraCFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, targetPosition), smoothFactor)

    Camera.CFrame = newCameraCFrame
end


-- Fly Logic
local function handleFly()
    if not flyEnabled then return end

    local moveDirection = Vector3.new(0, 0, 0)
    if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDirection = moveDirection + Camera.CFrame.LookVector end
    if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDirection = moveDirection - Camera.CFrame.LookVector end
    if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDirection = moveDirection + Camera.CFrame.RightVector * -1 end
    if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDirection = moveDirection + Camera.CFrame.RightVector end
    if UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveDirection = moveDirection + Vector3.new(0, 1, 0) end
    if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then moveDirection = moveDirection + Vector3.new(0, -1, 0) end

    -- Normalize and apply speed, ensure zero vector doesn't cause issues
    if moveDirection.Magnitude > 0 then
        flyVelocity = moveDirection.Unit * flySpeed
    else
        flyVelocity = Vector3.new(0, 0, 0)
    end
    HumanoidRootPart.AssemblyLinearVelocity = flyVelocity
end

-- Respawn at Death Logic
local lastPlayerPosition = nil
local deathLocation = nil

local function handleRespawnAtDeath()
    if not respawnAtDeath then return end

    if Humanoid.Health <= 0 then
        if not deathLocation then -- Just died
            deathLocation = HumanoidRootPart.Position
        end
        -- Keep checking if the player is still dead
    elseif Humanoid.Health > 0 and deathLocation then
        -- Player has respawned
        if (HumanoidRootPart.Position - deathLocation).Magnitude > 15 then -- Check if significantly moved from death location
            -- Player respawned somewhere else, reset
            deathLocation = nil
        else
            -- Player respawned very close to death location, force teleport
            HumanoidRootPart.CFrame = CFrame.new(deathLocation)
            deathLocation = nil
        end
    end
end


-- HumanoidRootPart Rotation Logic
local originalHRTPosition = nil
local touchTime = 0
local function handleHRTRotation()
    if not rotateHRT then return end

    local hrp = Character.HumanoidRootPart
    local hrpPosition = hrp.Position

    -- Detect if the HRT has been moved (touched)
    if not originalHRTPosition then
        originalHRTPosition = hrpPosition
    end

    if hrpPosition ~= originalHRTPosition then
        touchTime = tick()
        originalHRTPosition = hrpPosition
    end

    -- Check if it was recently touched
    if tick() - touchTime < 0.15 then -- Slightly longer grace period
        local target = getClosestTarget() -- Use the aimbot's target finding
        if target then
            local targetHRP = target.Character.HumanoidRootPart
            local lookAt = CFrame.new(hrpPosition, targetHRP.Position)
            hrp.CFrame = lookAt * CFrame.Angles(0, math.rad(90), 0) -- Rotate 90 degrees sideways for a fling effect
        else
            -- If no specific target, just spin it to simulate erratic movement
            hrp.CFrame = hrp.CFrame * CFrame.Angles(0, math.rad(20), 0) -- Faster spin
        end
    else
        originalHRTPosition = nil -- Reset if not touched recently
    end
end
